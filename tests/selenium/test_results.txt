from selenium import webdriver

from selenium.webdriver.chrome.service import Service

import time

class OpenBMCTestBase:
    def setup_driver(self):
        """Настройка драйвера для тестов"""
        chrome_options = Options()
        chrome_options.add_argument("--ignore-certificate-errors")
        chrome_options.add_argument("--ignore-ssl-errors")
        chrome_options.add_argument("--allow-running-insecure-content")
        chrome_options.add_argument("--disable-web-security")
        chrome_options.add_argument("--no-sandbox")
        chrome_options.add_argument("--disable-dev-shm-usage")
        
        self.driver = webdriver.Chrome(options=chrome_options)
        self.driver.implicitly_wait(10)
        self.driver.maximize_window()
        
        self.base_url = "https://127.0.0.1:2443"
        self.login_url = f"{self.base_url}/login"
        
        self.valid_username = "root"
        self.valid_password = "0penBmc"
        self.invalid_username = "invalid_user"
        self.invalid_password = "wrong_password"

    def teardown_driver(self):
        """Очистка после тестов"""
        if hasattr(self, 'driver') and self.driver:
            self.driver.quit()

    def login(self, username, password):
        """Вспомогательная функция для авторизации"""
        driver = self.driver
        print(f"Переход на страницу логина: {self.login_url}")
        driver.get(self.login_url)
        time.sleep(2)
        
        print(f"Текущий URL: {driver.current_url}")
        
        username_field = WebDriverWait(driver, 5).until(
            EC.presence_of_element_located((By.ID, "username"))
        )
        password_field = driver.find_element(By.ID, "password")
        login_button = driver.find_element(By.XPATH, '//button[@type="submit"]')
        
        username_field.clear()
        username_field.send_keys(username)
        password_field.clear()
        password_field.send_keys(password)
        print(f"Введены данные: {username}/{'*' * len(password)}")
        login_button.click()
        
        time.sleep(2)
        print(f"URL после логина: {driver.current_url}")

    def is_logged_in(self):
        """Проверка, что пользователь авторизован"""
        current_url = self.driver.current_url
        print(f"Проверка авторизации. Текущий URL: {current_url}")
        
        # Если мы не на странице логина - считаем что вошли
        if "/login" not in current_url:
            print("✓ Не на странице логина - считаем авторизованным")
            return True
        
        # Дополнительные проверки для страницы после логина
        try:
            # Ищем элементы, которые есть только после входа
            dashboard_elements = self.driver.find_elements(By.XPATH, 
                "//*[contains(text(), 'Dashboard') or contains(text(), 'System') or "
                "contains(text(), 'Overview') or contains(text(), 'Server') or "
                "contains(@class, 'dashboard') or contains(@class, 'navbar') or "
                "contains(@class, 'header')]"
            )
            if dashboard_elements:
                print("✓ Найдены элементы дашборда - авторизован")
                return True
        except:
            pass
        
        # Проверяем наличие навигации
        try:
            nav_elements = self.driver.find_elements(By.TAG_NAME, "nav")
            menu_elements = self.driver.find_elements(By.TAG_NAME, "menu")
            if nav_elements or menu_elements:
                print("✓ Найдена навигация - авторизован")
                return True
        except:
            pass
        
        print("✗ Не авторизован - остались на странице логина")
        return False


def authenticate_user(username, passkey):
    """Аутентификация пользователя в системе"""
    test_base = OpenBMCTestBase()
    try:
        test_base.setup_driver()
        test_base.login(username, passkey)
        
        # Проверяем авторизацию
        is_logged_in = test_base.is_logged_in()
        
        if is_logged_in:
            print("✓ Успешный вход подтвержден")
            return True
        else:
            # Проверяем ошибки
            try:
                error_elements = test_base.driver.find_elements(By.XPATH, 
                    '//*[contains(text(), "error") or contains(text(), "invalid") or contains(text(), "incorrect")]'
                )
                if error_elements:
                    print(f"✗ Ошибка входа: {error_elements[0].text}")
            except:
                pass
            return False
        
    except Exception as e:
        print(f"✗ Исключение при аутентификации: {e}")
        return False
    
    finally:
        test_base.teardown_driver()


def verify_account_lockout():
    """Проверка блокировки учетной записи после нескольких неудачных попыток"""
    test_base = OpenBMCTestBase()
    test_base.setup_driver()
    login_attempts = 3
    
    try:
        for current_attempt in range(1, login_attempts + 1):
            print(f"\nПопытка аутентификации {current_attempt}/{login_attempts}")
            
            test_base.driver.get(test_base.login_url)
            time.sleep(2)
            
            try:
                test_base.login('root', f'incorrect_pass_{current_attempt}')
            except Exception as e:
                print(f"Ошибка при логине: {e}")
                continue
            
            # Проверяем остались ли на странице логина
            if not test_base.is_logged_in():
                print("Неудачная попытка входа")
            else:
                print("Успешный вход (не ожидалось)")
                return False
        
        print("\nПробуем войти с правильными данными...")
        test_base.driver.get(test_base.login_url)
        time.sleep(2)
        
        test_base.login('root', '0penBmc')
        
        # Проверяем результат
        if test_base.is_logged_in():
            print("✗ Аккаунт НЕ заблокирован - вошли успешно")
            return False
        else:
            print("✓ Аккаунт заблокирован - не удалось войти")
            return True
            
    except Exception as e:
        print(f"✗ Ошибка: {e}")
        return False
    
    finally:
        test_base.teardown_driver()


def validate_hardware_inventory():
    """Проверка отображения инвентаризации оборудования"""
    test_base = OpenBMCTestBase()
    test_base.setup_driver()
    
    try:
        # Логинимся
        print("Вход в систему...")
        test_base.login('root', '0penBmc')
        time.sleep(2)

        # Проверяем авторизацию
        if not test_base.is_logged_in():
            print("✗ Не удалось войти в систему")
            return False

        print("✓ Успешный вход, поиск инвентаризации...")

        # Пробуем разные URL для инвентаризации
        inventory_urls = [
            f"{test_base.base_url}/#/hardware-status/inventory",
            f"{test_base.base_url}/#/inventory", 
            f"{test_base.base_url}/#/system/inventory",
            f"{test_base.base_url}/#/hardware/inventory",
            f"{test_base.base_url}/#/overview"
        ]
        
        for inventory_url in inventory_urls:
            print(f"Проверка URL: {inventory_url}")
            try:
                test_base.driver.get(inventory_url)
                time.sleep(2)
                
                # Проверяем авторизацию
                if not test_base.is_logged_in():
                    print("Потеряна авторизация, перелогиниваемся...")
                    test_base.login('root', '0penBmc')
                    time.sleep(1)
                    continue
                
                # Ищем элементы инвентаризации
                hardware_keywords = ["CPU", "Processor", "Memory", "RAM", "Hardware", "Inventory", "System"]
                
                for keyword in hardware_keywords:
                    elements = test_base.driver.find_elements(By.XPATH, f'//*[contains(text(), "{keyword}")]')
                    if elements:
                        print(f"✓ Найдены элементы инвентаризации: {keyword}")
                        for component in elements[:2]:
                            if component.text.strip():
                                print(f"   - {component.text[:50]}...")
                        return True
                        
            except Exception as e:
                print(f"Ошибка при проверке {inventory_url}: {e}")

        # Поиск через навигацию
        print("Поиск через навигацию...")
        nav_keywords = ["Inventory", "Hardware", "System", "Components"]
        for keyword in nav_keywords:
            try:
                elements = test_base.driver.find_elements(By.XPATH, f'//*[contains(text(), "{keyword}")]')
                for element in elements:
                    if element.is_displayed() and element.is_enabled():
                        print(f"Найден элемент: {element.text}")
                        try:
                            element.click()
                            time.sleep(2)
                            # Быстрая проверка
                            test_elements = test_base.driver.find_elements(By.XPATH, '//*[contains(text(), "CPU") or contains(text(), "Memory")]')
                            if test_elements:
                                print("✓ Инвентаризация найдена")
                                return True
                        except Exception as e:
                            print(f"Ошибка при клике: {e}")
            except:
                continue
        
        print("✗ Инвентаризация не найдена")
        return False
        
    except Exception as e:
        print(f"✗ Ошибка: {e}")
        return False
    
    finally:
        test_base.teardown_driver()


def check_power_operation_logs():
    """Проверка логов операций с питанием"""
    test_base = OpenBMCTestBase()
    test_base.setup_driver()
    
    try:
        # Логинимся
        print("Вход в систему...")
        test_base.login('root', '0penBmc')
        time.sleep(2)

        # Проверяем авторизацию
        if not test_base.is_logged_in():
            print("✗ Не удалось войти в систему")
            return False

        print("✓ Успешный вход, проверка операций...")

        # Пробуем найти кнопки питания
        power_operation_performed = False
        
        # Сначала на текущей странице
        power_buttons = test_base.driver.find_elements(By.XPATH, 
            '//button[contains(translate(text(), "ABCDEFGHIJKLMNOPQRSTUVWXYZ", "abcdefghijklmnopqrstuvwxyz"), "power") or '
            'contains(translate(text(), "ABCDEFGHIJKLMNOPQRSTUVWXYZ", "abcdefghijklmnopqrstuvwxyz"), "on")]'
        )
        
        if power_buttons:
            for btn in power_buttons:
                if btn.is_enabled():
                    print(f"Найдена кнопка: {btn.text}")
                    try:
                        original_url = test_base.driver.current_url
                        btn.click()
                        time.sleep(3)
                        power_operation_performed = True
                        print("✓ Операция питания выполнена")
                        test_base.driver.get(original_url)
                        time.sleep(1)
                        break
                    except Exception as e:
                        print(f"Ошибка при нажатии: {e}")

        # Если не нашли, пробуем URL
        if not power_operation_performed:
            power_urls = [
                f"{test_base.base_url}/#/operations/server-power-operations",
                f"{test_base.base_url}/#/power",
                f"{test_base.base_url}/#/operations/power"
            ]
            
            for power_url in power_urls:
                print(f"Переход: {power_url}")
                try:
                    test_base.driver.get(power_url)
                    time.sleep(2)
                    
                    if not test_base.is_logged_in():
                        test_base.login('root', '0penBmc')
                        time.sleep(1)
                        continue
                    
                    power_buttons = test_base.driver.find_elements(By.XPATH, '//button[contains(translate(text(), "ABCDEFGHIJKLMNOPQRSTUVWXYZ", "abcdefghijklmnopqrstuvwxyz"), "power")]')
                    if power_buttons:
                        for btn in power_buttons:
                            if btn.is_enabled():
                                btn.click()
                                time.sleep(3)
                                power_operation_performed = True
                                print("✓ Операция питания выполнена")
                                break
                        if power_operation_performed:
                            break
                except Exception as e:
                    print(f"Ошибка: {e}")

        # Проверяем логи
        print("Поиск логов...")
        log_urls = [
            f"{test_base.base_url}/#/logs/event-logs", 
            f"{test_base.base_url}/#/logs",
            f"{test_base.base_url}/#/system/logs"
        ]
        
        for log_url in log_urls:
            print(f"Проверка логов: {log_url}")
            try:
                test_base.driver.get(log_url)
                time.sleep(2)
                
                if not test_base.is_logged_in():
                    test_base.login('root', '0penBmc')
                    time.sleep(1)
                    continue
                
                # Ищем любые логи
                log_elements = test_base.driver.find_elements(By.XPATH, 
                    '//*[contains(text(), "Power") or contains(text(), "power") or '
                    'contains(text(), "Event") or contains(text(), "Log") or '
                    'contains(text(), "System") or contains(@class, "log")]'
                )
                
                if log_elements:
                    print(f"✓ Найдены логи: {len(log_elements)} элементов")
                    for entry in log_elements[:2]:
                        if entry.text.strip():
                            print(f"   - {entry.text[:80]}...")
                    return True
                    
            except Exception as e:
                print(f"Ошибка при проверке {log_url}: {e}")

        print("✗ Логи не найдены")
        return False

    except Exception as e:
        print(f"✗ Ошибка: {e}")
        return False
    
    finally:
        test_base.teardown_driver()

def run_simple():
    print("=" * 60)
    print("ЗАПУСК ТЕСТОВ OPENBMC")
    print("=" * 60)
    
    tests = [
        ("Успешная аутентификация", test_successful_authentication),
        ("Неудачная аутентификация", test_failed_authentication),
        ("Защита от блокировки", test_account_lockout_protection),
        ("Инвентаризация оборудования", test_hardware_inventory_display),
        ("Системные логи", test_system_log_entries),
    ]
    
    passed = 0
    failed = 0
    
    for test_name, test_func in tests:
        print(f"\n>>> ЗАПУСК: {test_name}")
        try:
            result = test_func()
            if result:
                print(f"✓ {test_name} - ПРОЙДЕН")
                passed += 1
            else:
                print(f"✗ {test_name} - ПРОВАЛЕН")
                failed += 1
        except Exception as e:
            print(f"✗ {test_name} - ОШИБКА: {e}")
            failed += 1
        print("-" * 50)
    
    print("\n" + "=" * 60)
    print("ИТОГИ ТЕСТИРОВАНИЯ")
    print("=" * 60)
    print(f"Всего тестов: {len(tests)}")
    print(f"Пройдено: {passed}")
    print(f"Провалено: {failed}")
    
    return failed


if __name__ == "__main__":
    print("Запуск тестов OpenBMC...")
    exit_code = run_simple()
    sys.exit(exit_code)
